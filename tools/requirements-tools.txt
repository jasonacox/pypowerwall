# Pinned build tools for generating protobuf Python bindings.
#
# These are DEVELOPER/CI tools only — not runtime dependencies.
# Install with: pip install -r tools/requirements-tools.txt
#
# grpcio-tools 1.81.x bundles a protoc that emits gencode tagged
# "Protobuf Python Version: 6.33.5" with a runtime version check, so the
# generated pb2 files require protobuf>=6.33.5 at runtime. These are pinned for
# reproducible, byte-stable output: CI runs tools/gen_proto.sh with these exact
# versions and verifies the result matches the committed pb2 files
# (git diff --exit-code). Unlike the old 4.25.x pin, this toolchain ships
# wheels for current Python versions (3.13+).
#
# If you upgrade these pins you MUST:
#   1. Run: bash tools/gen_proto.sh
#   2. Update the protobuf floor in requirements.txt and setup.py to match the
#      "Protobuf Python Version" comment in any regenerated pb2 file.
#   3. Consider whether the upgrade is a breaking change for library users.
grpcio-tools==1.81.1
protobuf==6.33.6
